```python
# apps/core/views.py — 替换 DemoGoodsListingSyncView

class GoodsListingView(APIView):
    """商品上货到目标平台"""
    permission_classes = _BUSINESS_API_PERMISSIONS

    @extend_schema(summary="商品上货")
    def post(self, request):
        goods_id = request.data.get("goods_id")
        platforms = request.data.get("platforms", [])
        if not goods_id or not platforms:
            return error_response(message="goods_id and platforms are required", status_code=400)

        goods = get_object_or_404(Product, id=goods_id)
        results = []
        for platform in platforms:
            try:
                # TODO: 调用平台API上架商品
                client = get_platform_client(platform)
                listing_id = client.create_listing(goods)
                results.append({"platform": platform, "status": "success", "external_id": listing_id})
            except Exception as exc:
                results.append({"platform": platform, "status": "failed", "reason": str(exc)})

        return success_response({"listing_results": results})
```
